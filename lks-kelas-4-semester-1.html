[tex]f(x) = 3x + 4 \\ f( - 1) = 3( - 1)+ 4 = - 3 + 4 = 1 \\ f( 0) = 3( 0)+ 4 = 0 + 4 = 4 \\ f( 1) = 3( 1)+ 4 = 3 + 4 = 7 \\ f( 2) = 3( 2)+ 4 = 6 + 4 = 10 \\ f( 3) = 3( 3)+ 4 = 9 + 4 = 13[/tex]
